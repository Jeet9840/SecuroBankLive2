document.addEventListener("DOMContentLoaded", function () { // Initialize balance if not present if (!localStorage.getItem("balance")) { localStorage.setItem("balance", 50000); // default balance } updateBalanceDisplay(); document.getElementById("transferForm") .addEventListener("submit", function (e) { e.preventDefault(); processTransfer(); }); }); function updateBalanceDisplay() { document.getElementById("currentBalance").innerText = localStorage.getItem("balance"); } function processTransfer() { let balance = parseFloat(localStorage.getItem("balance")); let amount = parseFloat(document.getElementById("amount").value); if (amount <= 0) { alert("Enter valid amount"); return; } if (amount > balance) { alert("Insufficient Balance!"); return; } // Deduct balance balance -= amount; localStorage.setItem("balance", balance); // Save transaction let transactions = JSON.parse(localStorage.getItem("transactions")) || []; transactions.push({ type: "Debit", beneficiary: document.getElementById("beneficiaryName").value, account: document.getElementById("accountNumber").value, amount: amount, date: new Date().toLocaleString() }); localStorage.setItem("transactions", JSON.stringify(transactions)); document.getElementById("transferMessage").innerText = "✅ Transfer Successful! ₹" + amount + " sent successfully."; document.getElementById("transferForm").reset(); updateBalanceDisplay(); }
